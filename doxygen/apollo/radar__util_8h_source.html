29  template <typename PointT>
30  static bool IsXyPointIn2dXyPolygon(const PointT &point,
31  const PolygonDType &polygon) {
32  bool in_poly = false;
33  double x1, x2, y1, y2;
34  int nr_poly_points = static_cast<int>(polygon.points.size());
35  // start with the last point to make the check last point<->first point the
36  // first one
37  double xold = polygon.points[nr_poly_points - 1].x;
38  double yold = polygon.points[nr_poly_points - 1].y;
39  for (int i = 0; i < nr_poly_points; i++) {
40  double xnew = polygon.points[i].x;
41  double ynew = polygon.points[i].y;
42  if (xnew > xold) {
43  x1 = xold;
44  x2 = xnew;
45  y1 = yold;
46  y2 = ynew;
47  } else {
48  x1 = xnew;
49  x2 = xold;
50  y1 = ynew;
51  y2 = yold;
52  }
53  if ((x1 < point.x) == (point.x <= x2) &&
54  (point.y - y1) * (x2 - x1) < (y2 - y1) * (point.x - x1)) {
55  in_poly = !in_poly;
56  }
57  xold = xnew;
58  yold = ynew;
59  }
60  return in_poly;
61  }
