70  template <std::size_t N>
71  std::pair<double, double> DeriveCartesianDeviation(
72  const std::size_t param_index) const {
73  auto gauss_points = common::math::GetGaussLegendrePoints<N>();
74  std::array<double, N> x = gauss_points.first;
75  std::array<double, N> w = gauss_points.second;
76 
77  std::pair<double, double> cartesian_deviation = {0.0, 0.0};
78  for (std::size_t i = 0; i < N; ++i) {
79  double r = 0.5 * x[i] + 0.5;
80  auto curr_theta = Evaluate(0, r * param_);
81  double derived_theta = DeriveTheta(param_index, r);
82 
83  cartesian_deviation.first +=
84  w[i] * (-std::sin(curr_theta)) * derived_theta;
85  cartesian_deviation.second +=
86  w[i] * std::cos(curr_theta) * derived_theta;
87  }
88 
89  cartesian_deviation.first *= param_ * 0.5;
90  cartesian_deviation.second *= param_ * 0.5;
91 
92  if (param_index == DELTA_S) {
93  for (std::size_t i = 0; i < N; ++i) {
94  double r = 0.5 * x[i] + 0.5;
95  auto theta_angle = Evaluate(0, r * param_);
96 
97  cartesian_deviation.first +=
98  0.5 * w[i] * std::cos(theta_angle);
99  cartesian_deviation.second +=
100  0.5 * w[i] * std::sin(theta_angle);
101  }
102  }
103  return cartesian_deviation;
104  }
