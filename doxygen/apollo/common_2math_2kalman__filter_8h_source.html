243 template <typename T, unsigned int XN, unsigned int ZN, unsigned int UN>
244 inline void KalmanFilter<T, XN, ZN, UN>::Predict(
245  const Eigen::Matrix<T, UN, 1> &u) {
246  CHECK(is_initialized_);
247  x_ = F_ * x_ + B_ * u;
248  P_ = F_ * P_ * F_.transpose() + Q_;
249 }

251 template <typename T, unsigned int XN, unsigned int ZN, unsigned int UN>
252 inline void KalmanFilter<T, XN, ZN, UN>::Correct(
253  const Eigen::Matrix<T, ZN, 1> &z) {
254  CHECK(is_initialized_);
255  y_ = z - H_ * x_;
256 
257  S_ = H_ * P_ * H_.transpose() + R_;
258 
259  K_ = P_ * H_.transpose() * PseudoInverse<T, ZN>(S_);
260 
261  x_ = x_ + K_ * y_;
262 
263  P_ = (Eigen::Matrix<T, XN, XN>::Identity() - K_ * H_) * P_;
264 }
