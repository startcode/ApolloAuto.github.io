176  // Step 1.
177  // For each row of the matrix, find the smallest element and subtract it
178  // from every element in its row. Go to Step 2.
179  void reduce_rows();
180 
181  // Step 2.
182  // Find a zero (Z) in the matrix. If there is no starred zero in its row
183  // or column, star Z. Repeat for every element in the matrix. Go to step 3.
184  // Note: profiling shows this method to use 9.2% of the CPU - the next
185  // slowest step takes 0.6%. I can't think of a way of speeding it up though.
186  void star_zeroes();
187 
188  // Step 3.
189  // Cover each column containing a starred zero. If all columns are
190  // covered, the starred zeros describe a complete set of unique assignments.
191  // In this case, terminate the algorithm. Otherwise, go to step 4.
192  void cover_starred_zeroes();
193 
194  // Step 4.
195  // Find a noncovered zero and prime it. If there is no starred zero in the
196  // row containing this primed zero, Go to Step 5. Otherwise, cover this row
197  // and uncover the column containing the starred zero. Continue in this manner
198  // until there are no uncovered zeros left, then go to Step 6.
199  void prime_zeroes();
200 
201  // Step 5.
202  // Construct a series of alternating primed and starred zeros as follows.
203  // Let Z0 represent the uncovered primed zero found in Step 4. Let Z1 denote
204  // the starred zero in the column of Z0 (if any). Let Z2 denote the primed
205  // zero in the row of Z1 (there will always be one). Continue until the
206  // series terminates at a primed zero that has no starred zero in its column.
207  // Unstar each starred zero of the series, star each primed zero of the
208  // series, erase all primes and uncover every line in the matrix. Return to
209  // Step 3.
210  void make_augmenting_path();
211 
212  // Step 6.
213  // Add the smallest uncovered value in the matrix to every element of each
214  // covered row, and subtract it from every element of each uncovered column.
215  // Return to Step 4 without altering any stars, primes, or covered lines.
216  void augment_path();
