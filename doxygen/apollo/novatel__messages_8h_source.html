495 inline ImuParameter get_imu_parameter(ImuType type) {
496  switch (type) {
497  case ImuType::IMAR_FSAS:
498  // 0.1 * (2 ** -8) * (math.pi / 180 / 3600), (0.05 * (2 ** -15)
499  return {1.893803441835e-9, 1.52587890625e-6, 200.0};
500 
501  case ImuType::ADIS16488:
502  // 720/2**31 deg/LSB, 200/2**31 m/s/LSB
503  return {5.8516723170686385e-09, 9.31322574615478515625e-8, 200.0};
504 
505  case ImuType::STIM300:
506  // 2**-21 deg/LSB, 2**-22 m/s/LSB
507  return {8.32237840649762e-09, 2.384185791015625e-07, 125.0};
508 
509  case ImuType::ISA100:
510  case ImuType::ISA100C:
511  // 1.0e-9 rad/LSB, 2.0e-8 m/s/LSB
512  return {1.0e-9, 2.0e-8, 200.0};
513 
514  case ImuType::ISA100_400HZ:
515  case ImuType::ISA100C_400HZ:
516  return {1.0e-9, 2.0e-8, 400.0};
517 
518  default:
519  return {0.0, 0.0, 0.0};
520  }
