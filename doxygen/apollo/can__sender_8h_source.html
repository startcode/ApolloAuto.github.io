276 template <typename SensorType>
277 void CanSender<SensorType>::PowerSendThreadFunc() {
278  CHECK_NOTNULL(can_client_);
279 
280  const int32_t INIT_PERIOD = 5000; // 5ms
281  int32_t delta_period = INIT_PERIOD;
282  int32_t new_delta_period = INIT_PERIOD;
283 
284  int64_t tm_start = 0;
285  int64_t tm_end = 0;
286  int64_t sleep_interval = 0;
287 
288  AINFO << "Can client sender thread starts.";
289 
290  while (is_running_) {
291  tm_start = common::time::AsInt64<micros>(Clock::Now());
292  new_delta_period = INIT_PERIOD;
293 
294  for (auto &message : send_messages_) {
295  bool need_send = NeedSend(message, delta_period);
296  message.UpdateCurrPeriod(delta_period);
297  new_delta_period = std::min(new_delta_period, message.curr_period());
298 
299  if (!need_send) {
300  continue;
301  }
302  std::vector<CanFrame> can_frames;
303  CanFrame can_frame = message.CanFrame();
304  can_frames.push_back(can_frame);
305  if (can_client_->SendSingleFrame(can_frames) != ErrorCode::OK) {
306  AERROR << "Send msg failed:" << can_frame.CanFrameString();
307  }
308  if (enable_log()) {
309  ADEBUG << "send_can_frame#" << can_frame.CanFrameString();
310  }
311  }
312  delta_period = new_delta_period;
313  tm_end = common::time::AsInt64<micros>(Clock::Now());
314  sleep_interval = delta_period - (tm_end - tm_start);
315 
316  if (sleep_interval > 0) {
317  std::this_thread::sleep_for(std::chrono::microseconds(sleep_interval));
318  } else {
319  // do not sleep
320  AWARN << "Too much time for calculation: " << tm_end - tm_start
321  << "us is more than minimum period: " << delta_period << "us";
322  }
323  }
324  AINFO << "Can client sender thread stopped!";
325 }
