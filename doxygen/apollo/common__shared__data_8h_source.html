156 template <class M>
157 void CommonSharedData<M>::RemoveStaleData() {
158  MutexLock lock(&mutex_);
159  const uint64_t now = ::time(NULL);
160  bool has_change = false;
161  for (auto iter = data_added_time_map_.begin();
162  iter != data_added_time_map_.end();) {
163  if (now - iter->second > FLAGS_shared_data_stale_time) {
164  const size_t erase_cnt = data_map_.erase(iter->first);
165  if (erase_cnt != 1u) {
166  AWARN << "_data_map erase cnt:" << erase_cnt << " key:" << iter->first;
167  return;
168  }
169  iter = data_added_time_map_.erase(iter);
170  ++stat_.remove_cnt;
171  has_change = true;
172  } else {
173  ++iter;
174  }
175  }
176  if (has_change) {
177  AINFO << "SharedData remove_stale_data name:" << name() << " stat:["
178  << stat_.ToString() << "]";
179  }
180 }
