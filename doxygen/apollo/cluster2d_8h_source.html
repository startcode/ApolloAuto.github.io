81  void Cluster(const caffe::Blob<float>& category_pt_blob,
82  const caffe::Blob<float>& instance_pt_blob,
83  const apollo::perception::pcl_util::PointCloudPtr& pc_ptr,
84  const apollo::perception::pcl_util::PointIndices& valid_indices,
85  float objectness_thresh, bool use_all_grids_for_clustering) {
86  const float* category_pt_data = category_pt_blob.cpu_data();
87  const float* instance_pt_x_data = instance_pt_blob.cpu_data();
88  const float* instance_pt_y_data =
89  instance_pt_blob.cpu_data() + instance_pt_blob.offset(0, 1);
90 
91  pc_ptr_ = pc_ptr;
92  std::vector<std::vector<Node>> nodes(rows_,
93  std::vector<Node>(cols_, Node()));
94 
95  // map points into grids
96  size_t tot_point_num = pc_ptr_->size();
97  valid_indices_in_pc_ = &(valid_indices.indices);
98  CHECK_LE(valid_indices_in_pc_->size(), tot_point_num);
99  point2grid_.assign(valid_indices_in_pc_->size(), -1);
100 
101  for (size_t i = 0; i < valid_indices_in_pc_->size(); ++i) {
102  int point_id = valid_indices_in_pc_->at(i);
103  CHECK_GE(point_id, 0);
104  CHECK_LT(point_id, static_cast<int>(tot_point_num));
105  const auto& point = pc_ptr_->points[point_id];
106  // * the coordinates of x and y have been exchanged in feature generation
107  // step,
108  // so we swap them back here.
109  int pos_x = F2I(point.y, range_, inv_res_x_); // col
110  int pos_y = F2I(point.x, range_, inv_res_y_); // row
111  if (IsValidRowCol(pos_y, pos_x)) {
112  // get grid index and count point number for corresponding node
113  point2grid_[i] = RowCol2Grid(pos_y, pos_x);
114  nodes[pos_y][pos_x].point_num++;
115  }
116  }
117 
118  // construct graph with center offset prediction and objectness
119  for (int row = 0; row < rows_; row++) {
120  for (int col = 0; col < cols_; col++) {
121  int grid = RowCol2Grid(row, col);
122  Node* node = &nodes[row][col];
123  apollo::perception::DisjointSetMakeSet(node);
124  node->is_object =
125  (use_all_grids_for_clustering || nodes[row][col].point_num > 0) &&
126  (*(category_pt_data + grid) >= objectness_thresh);
127  int center_row = std::round(row + instance_pt_x_data[grid] * scale_);
128  int center_col = std::round(col + instance_pt_y_data[grid] * scale_);
129  center_row = std::min(std::max(center_row, 0), rows_ - 1);
130  center_col = std::min(std::max(center_col, 0), cols_ - 1);
131  node->center_node = &nodes[center_row][center_col];
132  }
133  }
134 
135  // traverse nodes
136  for (int row = 0; row < rows_; row++) {
137  for (int col = 0; col < cols_; col++) {
138  Node* node = &nodes[row][col];
139  if (node->is_object && node->traversed == 0) {
140  Traverse(node);
141  }
142  }
143  }
144  for (int row = 0; row < rows_; row++) {
145  for (int col = 0; col < cols_; col++) {
146  Node* node = &nodes[row][col];
147  if (!node->is_center) {
148  continue;
149  }
150  for (int row2 = row - 1; row2 <= row + 1; row2++) {
151  for (int col2 = col - 1; col2 <= col + 1; col2++) {
152  if ((row2 == row || col2 == col) && IsValidRowCol(row2, col2)) {
153  Node* node2 = &nodes[row2][col2];
154  if (node2->is_center) {
155  apollo::perception::DisjointSetUnion(node, node2);
156  }
157  }
158  }
159  }
160  }
161  }
162 
163  int count_obstacles = 0;
164  obstacles_.clear();
165  id_img_.assign(grids_, -1);
166  for (int row = 0; row < rows_; row++) {
167  for (int col = 0; col < cols_; col++) {
168  Node* node = &nodes[row][col];
169  if (!node->is_object) {
170  continue;
171  }
172  Node* root = apollo::perception::DisjointSetFind(node);
173  if (root->obstacle_id < 0) {
174  root->obstacle_id = count_obstacles++;
175  CHECK_EQ(static_cast<int>(obstacles_.size()), count_obstacles - 1);
176  obstacles_.push_back(Obstacle());
177  }
178  int grid = RowCol2Grid(row, col);
179  CHECK_GE(root->obstacle_id, 0);
180  id_img_[grid] = root->obstacle_id;
181  obstacles_[root->obstacle_id].grids.push_back(grid);
182  }
183  }
184  CHECK_EQ(static_cast<int>(count_obstacles), obstacles_.size());
185  }

187  void Filter(const caffe::Blob<float>& confidence_pt_blob,
188  const caffe::Blob<float>& height_pt_blob) {
189  const float* confidence_pt_data = confidence_pt_blob.cpu_data();
190  const float* height_pt_data = height_pt_blob.cpu_data();
191  for (size_t obstacle_id = 0; obstacle_id < obstacles_.size();
192  obstacle_id++) {
193  Obstacle* obs = &obstacles_[obstacle_id];
194  CHECK_GT(obs->grids.size(), 0);
195  double score = 0.0;
196  double height = 0.0;
197  for (int grid : obs->grids) {
198  score += static_cast<double>(confidence_pt_data[grid]);
199  height += static_cast<double>(height_pt_data[grid]);
200  }
201  obs->score = score / static_cast<double>(obs->grids.size());
202  obs->height = height / static_cast<double>(obs->grids.size());
203  obs->cloud.reset(new apollo::perception::pcl_util::PointCloud);
204  }
205  }

207  void Classify(const caffe::Blob<float>& classify_pt_blob) {
208  const float* classify_pt_data = classify_pt_blob.cpu_data();
209  int num_classes = classify_pt_blob.channels();
210  CHECK_EQ(num_classes, MAX_META_TYPE);
211  for (size_t obs_id = 0; obs_id < obstacles_.size(); obs_id++) {
212  Obstacle* obs = &obstacles_[obs_id];
213  for (size_t grid_id = 0; grid_id < obs->grids.size(); grid_id++) {
214  int grid = obs->grids[grid_id];
215  for (int k = 0; k < num_classes; k++) {
216  obs->meta_type_probs[k] += classify_pt_data[k * grids_ + grid];
217  }
218  }
219  int meta_type_id = 0;
220  for (int k = 0; k < num_classes; k++) {
221  obs->meta_type_probs[k] /= obs->grids.size();
222  if (obs->meta_type_probs[k] > obs->meta_type_probs[meta_type_id]) {
223  meta_type_id = k;
224  }
225  }
226  obs->meta_type = static_cast<MetaType>(meta_type_id);
227  }
228  }

230  void GetObjects(const float confidence_thresh, const float height_thresh,
231  const int min_pts_num, std::vector<ObjectPtr>* objects) {
232  CHECK(valid_indices_in_pc_ != nullptr);
233 
234  for (size_t i = 0; i < point2grid_.size(); ++i) {
235  int grid = point2grid_[i];
236  if (grid < 0) {
237  continue;
238  }
239 
240  CHECK_GE(grid, 0);
241  CHECK_LT(grid, grids_);
242  int obstacle_id = id_img_[grid];
243 
244  int point_id = valid_indices_in_pc_->at(i);
245  CHECK_GE(point_id, 0);
246  CHECK_LT(point_id, static_cast<int>(pc_ptr_->size()));
247 
248  if (obstacle_id >= 0 &&
249  obstacles_[obstacle_id].score >= confidence_thresh) {
250  if (height_thresh < 0 ||
251  pc_ptr_->points[point_id].z <=
252  obstacles_[obstacle_id].height + height_thresh) {
253  obstacles_[obstacle_id].cloud->push_back(pc_ptr_->points[point_id]);
254  }
255  }
256  }
257 
258  for (size_t obstacle_id = 0; obstacle_id < obstacles_.size();
259  obstacle_id++) {
260  Obstacle* obs = &obstacles_[obstacle_id];
261  if (static_cast<int>(obs->cloud->size()) < min_pts_num) {
262  continue;
263  }
264  apollo::perception::ObjectPtr out_obj(new apollo::perception::Object);
265  out_obj->cloud = obs->cloud;
266  out_obj->score = obs->score;
267  out_obj->type = GetObjectType(obs->meta_type);
268  out_obj->type_probs = GetObjectTypeProbs(obs->meta_type_probs);
269  objects->push_back(out_obj);
270  }
271  }

312  void Traverse(Node* x) {
313  std::vector<Node*> p;
314  p.clear();
315  while (x->traversed == 0) {
316  p.push_back(x);
317  x->traversed = 2;
318  x = x->center_node;
319  }
320  if (x->traversed == 2) {
321  for (int i = static_cast<int>(p.size()) - 1; i >= 0 && p[i] != x; i--) {
322  p[i]->is_center = true;
323  }
324  x->is_center = true;
325  }
326  for (size_t i = 0; i < p.size(); i++) {
327  Node* y = p[i];
328  y->traversed = 1;
329  y->parent = x->parent;
330  }
331  }

334  switch (meta_type_id) {

352  std::vector<float> GetObjectTypeProbs(
