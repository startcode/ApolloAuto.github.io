18 //
19 // CODE SAMPLE:
20 // you can use such code to access your parameters:
21 //
22 // #include "lib/config_manager/config_manager.h"
23 //
24 // ConfigManager* config_manager = ConfigManager::instance();
25 //
26 // string model_name = "FrameClassifier";
27 // const ModelConfig* model_config = NULL;
28 // if (!config_manager->GetModelConfig(model_name, &model_config)) {
29 // XLOG(ERROR) << "not found model: " << model_name;
30 // return false;
31 // }
32 //
33 // int int_value = 0;
34 // if (!model_config->GetValue("my_param_name", &int_value)) {
35 // XLOG(ERROR) << "my_param_name not found."
36 // return false;
37 // }
38 // using int_value....
39 //
40 //
41 // CONFIG FORMAT
42 //
43 // First you should define file: conf/config_manager.config,
44 // you can set the path by gflags
45 // --config_manager_path=conf/config_manager.config
46 //
47 // file content like as:
48 // define all model config paths.
49 // ############################################
50 //
51 // model_config_path: "./conf/frame_classifier.config"
52 // model_config_path: "./conf/track_classifier.config"
53 //
54 // ############################################
55 //
56 // one line identify one model parameter config path.
57 // ModelConfig config file like as:
58 // file: ./conf/frame_classifier.config
59 // #############################################
60 //
61 // model_configs {
62 // # FrameClassifier model.
63 // name: "FrameClassifier"
64 // version: "1.0.0"
65 // integer_params {
66 // name: "threshold1"
67 // value: 1
