180 template <typename SensorType>
181 void MessageManager<SensorType>::Parse(const uint32_t message_id,
182  const uint8_t *data, int32_t length) {
183  ProtocolData<SensorType> *protocol_data =
184  GetMutableProtocolDataById(message_id);
185  if (protocol_data == nullptr) {
186  return;
187  }
188  {
189  std::lock_guard<std::mutex> lock(sensor_data_mutex_);
190  protocol_data->Parse(data, length, &sensor_data_);
191  }
192  received_ids_.insert(message_id);
193  // check if need to check period
194  const auto it = check_ids_.find(message_id);
195  if (it != check_ids_.end()) {
196  const int64_t time = apollo::common::time::AsInt64<micros>(Clock::Now());
197  it->second.real_period = time - it->second.last_time;
198  // if period 1.5 large than base period, inc error_count
199  const double period_multiplier = 1.5;
200  if (it->second.real_period > (it->second.period * period_multiplier)) {
201  it->second.error_count += 1;
202  } else {
203  it->second.error_count = 0;
204  }
205  it->second.last_time = time;
206  }
207 }
