164  bool SplitToSubNodes(const std::vector<ObjectPtr> &objects,
165  const AABoxKDTreeParams &params) {
166  if (params.max_depth >= 0 && depth_ >= params.max_depth) {
167  return false;
168  }
169  if (static_cast<int>(objects.size()) <= std::max(1, params.max_leaf_size)) {
170  return false;
171  }
172  if (params.max_leaf_dimension >= 0.0 &&
173  std::max(max_x_ - min_x_, max_y_ - min_y_) <=
174  params.max_leaf_dimension) {
175  return false;
176  }
177  return true;
178  }

180  double LowerDistanceSquareToPoint(const Vec2d &point) const {
181  double dx = 0.0;
182  if (point.x() < min_x_) {
183  dx = min_x_ - point.x();
184  } else if (point.x() > max_x_) {
185  dx = point.x() - max_x_;
186  }
187  double dy = 0.0;
188  if (point.y() < min_y_) {
189  dy = min_y_ - point.y();
190  } else if (point.y() > max_y_) {
191  dy = point.y() - max_y_;
192  }
193  return dx * dx + dy * dy;
194  }

196  double UpperDistanceSquareToPoint(const Vec2d &point) const {
197  const double dx =
198  (point.x() > mid_x_ ? (point.x() - min_x_) : (point.x() - max_x_));
199  const double dy =
200  (point.y() > mid_y_ ? (point.y() - min_y_) : (point.y() - max_y_));
201  return dx * dx + dy * dy;
202  }

216  void GetObjectsInternal(const Vec2d &point, const double distance,
217  const double distance_sqr,
218  std::vector<ObjectPtr> *const result_objects) const {
219  if (LowerDistanceSquareToPoint(point) > distance_sqr) {
220  return;
221  }
222  if (UpperDistanceSquareToPoint(point) <= distance_sqr) {
223  GetAllObjects(result_objects);
224  return;
225  }
226  const double pvalue = (partition_ == PARTITION_X ? point.x() : point.y());
227  if (pvalue < partition_position_) {
228  const double limit = pvalue + distance;
229  for (int i = 0; i < num_objects_; ++i) {
230  if (objects_sorted_by_min_bound_[i] > limit) {
231  break;
232  }
233  ObjectPtr object = objects_sorted_by_min_[i];
234  if (object->DistanceSquareTo(point) <= distance_sqr) {
235  result_objects->push_back(object);
236  }
237  }
238  } else {
239  const double limit = pvalue - distance;
240  for (int i = 0; i < num_objects_; ++i) {
241  if (objects_sorted_by_max_bound_[i] < limit) {
242  break;
243  }
244  ObjectPtr object = objects_sorted_by_max_[i];
245  if (object->DistanceSquareTo(point) <= distance_sqr) {
246  result_objects->push_back(object);
247  }
248  }
249  }
250  if (left_subnode_ != nullptr) {
251  left_subnode_->GetObjectsInternal(point, distance, distance_sqr,
252  result_objects);
253  }
254  if (right_subnode_ != nullptr) {
255  right_subnode_->GetObjectsInternal(point, distance, distance_sqr,
256  result_objects);
257  }
258  }

260  void GetNearestObjectInternal(const Vec2d &point,
261  double *const min_distance_sqr,
262  ObjectPtr *const nearest_object) const {
263  if (LowerDistanceSquareToPoint(point) >= *min_distance_sqr - kMathEpsilon) {
264  return;
265  }
266  const double pvalue = (partition_ == PARTITION_X ? point.x() : point.y());
267  const bool search_left_first = (pvalue < partition_position_);
268  if (search_left_first) {
269  if (left_subnode_ != nullptr) {
270  left_subnode_->GetNearestObjectInternal(point, min_distance_sqr,
271  nearest_object);
272  }
273  } else {
274  if (right_subnode_ != nullptr) {
275  right_subnode_->GetNearestObjectInternal(point, min_distance_sqr,
276  nearest_object);
277  }
278  }
279  if (*min_distance_sqr <= kMathEpsilon) {
280  return;
281  }
282 
283  if (search_left_first) {
284  for (int i = 0; i < num_objects_; ++i) {
285  const double bound = objects_sorted_by_min_bound_[i];
286  if (bound > pvalue && Square(bound - pvalue) > *min_distance_sqr) {
287  break;
288  }
289  ObjectPtr object = objects_sorted_by_min_[i];
290  const double distance_sqr = object->DistanceSquareTo(point);
291  if (distance_sqr < *min_distance_sqr) {
292  *min_distance_sqr = distance_sqr;
293  *nearest_object = object;
294  }
295  }
296  } else {
297  for (int i = 0; i < num_objects_; ++i) {
298  const double bound = objects_sorted_by_max_bound_[i];
299  if (bound < pvalue && Square(bound - pvalue) > *min_distance_sqr) {
300  break;
301  }
302  ObjectPtr object = objects_sorted_by_max_[i];
303  const double distance_sqr = object->DistanceSquareTo(point);
304  if (distance_sqr < *min_distance_sqr) {
305  *min_distance_sqr = distance_sqr;
306  *nearest_object = object;
307  }
308  }
309  }
310  if (*min_distance_sqr <= kMathEpsilon) {
311  return;
312  }
313  if (search_left_first) {
314  if (right_subnode_ != nullptr) {
315  right_subnode_->GetNearestObjectInternal(point, min_distance_sqr,
316  nearest_object);
317  }
318  } else {
319  if (left_subnode_ != nullptr) {
320  left_subnode_->GetNearestObjectInternal(point, min_distance_sqr,
321  nearest_object);
322  }
323  }
324  }

326  void ComputeBoundary(const std::vector<ObjectPtr> &objects) {
327  min_x_ = std::numeric_limits<double>::infinity();
328  min_y_ = std::numeric_limits<double>::infinity();
329  max_x_ = -std::numeric_limits<double>::infinity();
330  max_y_ = -std::numeric_limits<double>::infinity();
331  for (ObjectPtr object : objects) {
332  min_x_ = std::min(min_x_, object->aabox().min_x());
333  max_x_ = std::max(max_x_, object->aabox().max_x());
334  min_y_ = std::min(min_y_, object->aabox().min_y());
335  max_y_ = std::max(max_y_, object->aabox().max_y());
336  }
337  mid_x_ = (min_x_ + max_x_) / 2.0;
338  mid_y_ = (min_y_ + max_y_) / 2.0;
339  }
340 
341  void ComputePartition() {
342  if (max_x_ - min_x_ >= max_y_ - min_y_) {
343  partition_ = PARTITION_X;
344  partition_position_ = (min_x_ + max_x_) / 2.0;
345  } else {
346  partition_ = PARTITION_Y;
347  partition_position_ = (min_y_ + max_y_) / 2.0;
348  }
349  }

351  void PartitionObjects(const std::vector<ObjectPtr> &objects,
352  std::vector<ObjectPtr> *const left_subnode_objects,
353  std::vector<ObjectPtr> *const right_subnode_objects) {
354  left_subnode_objects->clear();
355  right_subnode_objects->clear();
356  std::vector<ObjectPtr> other_objects;
357  if (partition_ == PARTITION_X) {
358  for (ObjectPtr object : objects) {
359  if (object->aabox().max_x() <= partition_position_) {
360  left_subnode_objects->push_back(object);
361  } else if (object->aabox().min_x() >= partition_position_) {
362  right_subnode_objects->push_back(object);
363  } else {
364  other_objects.push_back(object);
365  }
366  }
367  } else {
368  for (ObjectPtr object : objects) {
369  if (object->aabox().max_y() <= partition_position_) {
370  left_subnode_objects->push_back(object);
371  } else if (object->aabox().min_y() >= partition_position_) {
372  right_subnode_objects->push_back(object);
373  } else {
374  other_objects.push_back(object);
375  }
376  }
377  }
378  InitObjects(other_objects);
379  }
