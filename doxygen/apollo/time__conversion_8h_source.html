33 // There's two concept about 'time'.
34 // 1. Unix time : It counts seconds since Jan 1,1970.
35 // Unix time doesn't include the leap seconds.
36 //
37 // 2. GPS time : It counts seconds since Jan 6,1980.
38 // GPS time does include the leap seconds.
39 //
40 // Leap seconds was added in every a few years, See details:

73 // We build a lookup table to describe relationship that between UTC and
74 // Leap_seconds.
75 //
76 // Column1: UTC time diff (second).
77 // Shell Example:
78 // date +%s -d"Jan 1, 2017 00:00:00"
79 // return 1483257600.
80 //
81 // date +%s -d"Jan 1, 1970 00:00:00"
82 // return 28800.
83 //
84 // The diff between 1970/01/01 and 2017/01/01 is 1483228800.
85 // (1483257600-28800)
86 //
87 // Column2: Leap seconds diff with GPS basetime(Jan 6,1980).
88 // We Know that The leap_seconds have been already added 37 seconds
89 // util now(2017).
90 // So we can calculate leap_seconds diff between GPS (from Jan 6,1980)
91 // and UTC time.
92 
93 // calc with the formula.
94 static const int32_t LEAP_SECONDS[][2] = {
95  {1483228800, 18}, // 2017/01/01
96  {1435708800, 17}, // 2015/07/01
97  {1341100800, 16}, // 2012/07/01
98  {1230768000, 15}, // 2009/01/01
99  {1136073600, 14}, // 2006/01/01
100  {915148800, 13}, // 1999/01/01
101  {867711600, 12}, // 1997/07/01
102  {820480320, 11}, // 1996/01/01 ;)
103  //....
104  //..
105  // etc.
106 };

108 // seconds that UNIX time afront of GPS, without leap seconds.

116 const int32_t GPS_AND_SYSTEM_DIFF_SECONDS = 315964800;

118 template <typename T>
119 T unix2gps(T unix_seconds) {
120  for (size_t i = 0; i < array_size(LEAP_SECONDS); ++i) {
121  if (unix_seconds >= LEAP_SECONDS[i][0]) {
122  return unix_seconds - (GPS_AND_SYSTEM_DIFF_SECONDS - LEAP_SECONDS[i][1]);
123  }
124  }
125  return static_cast<T>(0);
126 }

128 template <typename T>
129 T gps2unix(T gps_seconds) {
130  for (size_t i = 0; i < array_size(LEAP_SECONDS); ++i) {
131  T result = gps_seconds + (GPS_AND_SYSTEM_DIFF_SECONDS - LEAP_SECONDS[i][1]);
132  if (result >= LEAP_SECONDS[i][0]) {
133  return result;
134  }
135  }
136  return static_cast<T>(0);
137 }
