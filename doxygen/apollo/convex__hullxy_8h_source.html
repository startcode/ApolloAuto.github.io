104  void PerformReconstruction2dxy(PointCloudPtr hull,
105  std::vector<pcl::Vertices> *polygons,
106  bool fill_polygon_data = false) {
107  int dimension = 2;
108 
109  // True if qhull should free points in qh_freeqhull() or reallocation
110  boolT ismalloc = True;
111  // output from qh_produce_output(), use NULL to skip qh_produce_output()
112  FILE *outfile = NULL;
113 
114 #ifndef HAVE_QHULL_2011
115  if (compute_area_) {
116  outfile = stderr;
117  }
118 #endif
119 
120  // option flags for qhull, see qh_opt.htm
121  const char *flags = qhull_flags.c_str();
122  // error messages from qhull code
123  FILE *errfile = stderr;
124 
125  // Array of coordinates for each point
126  coordT *points = reinterpret_cast<coordT *>(
127  calloc(indices_->size() * dimension, sizeof(coordT)));
128  if (points == NULL) {
129  hull->points.resize(0);
130  hull->width = hull->height = 0;
131  polygons->resize(0);
132  return;
133  }
134 
135  // Build input data, using appropriate projection
136  int j = 0;
137  for (size_t i = 0; i < indices_->size(); ++i, j += dimension) {
138  points[j + 0] = static_cast<coordT>(input_->points[(*indices_)[i]].x);
139  points[j + 1] = static_cast<coordT>(input_->points[(*indices_)[i]].y);
140  }
141 
142  // Compute convex hull
143  int exitcode =
144  qh_new_qhull(dimension, static_cast<int>(indices_->size()), points,
145  ismalloc, const_cast<char *>(flags), outfile, errfile);
146 #ifdef HAVE_QHULL_2011
147  if (compute_area_) {
148  qh_prepare_output();
149  }
150 #endif
151 
152  // 0 if no error from qhull or it doesn't find any vertices
153  if (exitcode != 0 || qh num_vertices == 0) {
154  PCL_ERROR(
155  "[pcl::%s::performReconstrution2D] "
156  "ERROR: qhull was unable to compute "
157  "a convex hull for the given point "
158  "cloud (%lu)!\n",
159  getClassName().c_str(), indices_->size());
160 
161  hull->points.resize(0);
162  hull->width = hull->height = 0;
163  polygons->resize(0);
164 
165  qh_freeqhull(!qh_ALL);
166  int curlong, totlong;
167  qh_memfreeshort(&curlong, &totlong);
168  return;
169  }
170 
171  // Qhull returns the area in volume for 2D
172  if (compute_area_) {
173  total_area_ = qh totvol;
174  total_volume_ = 0.0;
175  }
176 
177  int num_vertices = qh num_vertices;
178  hull->points.resize(num_vertices);
179  memset(&hull->points[0], static_cast<int>(hull->points.size()),
180  sizeof(PointInT));
181 
182  vertexT *vertex;
183  int i = 0;
184 
185  std::vector<std::pair<int, Eigen::Vector4f>,
186  Eigen::aligned_allocator<std::pair<int, Eigen::Vector4f>>>
187  idx_points(num_vertices);
188  idx_points.resize(hull->points.size());
189  memset(&idx_points[0], static_cast<int>(hull->points.size()),
190  sizeof(std::pair<int, Eigen::Vector4f>));
191 
192  FORALLvertices {
193  hull->points[i] = input_->points[(*indices_)[qh_pointid(vertex->point)]];
194  idx_points[i].first = qh_pointid(vertex->point);
195  ++i;
196  }
197 
198  // Sort
199  Eigen::Vector4f centroid;
200  pcl::compute3DCentroid(*hull, centroid);
201  for (size_t j = 0; j < hull->points.size(); j++) {
202  idx_points[j].second[0] = hull->points[j].x - centroid[0];
203  idx_points[j].second[1] = hull->points[j].y - centroid[1];
204  }
205 
206  std::sort(idx_points.begin(), idx_points.end(), pcl::comparePoints2D);
207 
208  polygons->resize(1);
209  (*polygons)[0].vertices.resize(hull->points.size());
210 
211  for (int j = 0; j < static_cast<int>(hull->points.size()); j++) {
212  hull->points[j] = input_->points[(*indices_)[idx_points[j].first]];
213  (*polygons)[0].vertices[j] = static_cast<unsigned int>(j);
214  }
215 
216  qh_freeqhull(!qh_ALL);
217  int curlong, totlong;
218  qh_memfreeshort(&curlong, &totlong);
219 
220  hull->width = static_cast<uint32_t>(hull->points.size());
221  hull->height = 1;
222  hull->is_dense = false;
223  return;
224  }
