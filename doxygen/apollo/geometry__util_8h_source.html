91 void GetCloudMinMax3D(typename pcl::PointCloud<PointT>::Ptr cloud,
92  Eigen::Vector4f* min_point, Eigen::Vector4f* max_point) {
93  Eigen::Vector4f& min_pt = *min_point;
94  Eigen::Vector4f& max_pt = *max_point;
95  min_pt[0] = min_pt[1] = min_pt[2] = FLT_MAX;
96  max_pt[0] = max_pt[1] = max_pt[2] = -FLT_MAX;
97  if (cloud->is_dense) {
98  for (size_t i = 0; i < cloud->points.size(); ++i) {
99  min_pt[0] = std::min(min_pt[0], cloud->points[i].x);
100  max_pt[0] = std::max(max_pt[0], cloud->points[i].x);
101  min_pt[1] = std::min(min_pt[1], cloud->points[i].y);
102  max_pt[1] = std::max(max_pt[1], cloud->points[i].y);
103  min_pt[2] = std::min(min_pt[2], cloud->points[i].z);
104  max_pt[2] = std::max(max_pt[2], cloud->points[i].z);
105  }
106  } else {
107  for (size_t i = 0; i < cloud->points.size(); ++i) {
108  if (!pcl_isfinite(cloud->points[i].x) ||
109  !pcl_isfinite(cloud->points[i].y) ||
110  !pcl_isfinite(cloud->points[i].z)) {
111  continue;
112  }
113  min_pt[0] = std::min(min_pt[0], cloud->points[i].x);
114  max_pt[0] = std::max(max_pt[0], cloud->points[i].x);
115  min_pt[1] = std::min(min_pt[1], cloud->points[i].y);
116  max_pt[1] = std::max(max_pt[1], cloud->points[i].y);
117  min_pt[2] = std::min(min_pt[2], cloud->points[i].z);
118  max_pt[2] = std::max(max_pt[2], cloud->points[i].z);
119  }
120  }

123 template <typename PointT>
124 void ComputeBboxSizeCenter(typename pcl::PointCloud<PointT>::Ptr cloud,
125  const Eigen::Vector3d& direction,
126  Eigen::Vector3d* size, Eigen::Vector3d* center) {
127  Eigen::Vector3d dir(direction[0], direction[1], 0);
128  dir.normalize();
129  Eigen::Vector3d ortho_dir(-dir[1], dir[0], 0.0);
130 
131  Eigen::Vector3d z_dir(dir.cross(ortho_dir));
132  Eigen::Vector3d min_pt(DBL_MAX, DBL_MAX, DBL_MAX);
133  Eigen::Vector3d max_pt(-DBL_MAX, -DBL_MAX, -DBL_MAX);
134 
135  Eigen::Vector3d loc_pt;
136  for (int i = 0; i < cloud->size(); i++) {
137  Eigen::Vector3d pt = Eigen::Vector3d(cloud->points[i].x, cloud->points[i].y,
138  cloud->points[i].z);
139  loc_pt[0] = pt.dot(dir);
140  loc_pt[1] = pt.dot(ortho_dir);
141  loc_pt[2] = pt.dot(z_dir);
142  for (int j = 0; j < 3; j++) {
143  min_pt[j] = std::min(min_pt[j], loc_pt[j]);
144  max_pt[j] = std::max(max_pt[j], loc_pt[j]);
145  }
146  }
147  *size = max_pt - min_pt;
148  *center = dir * ((max_pt[0] + min_pt[0]) * 0.5) +
149  ortho_dir * ((max_pt[1] + min_pt[1]) * 0.5) + z_dir * min_pt[2];
150 }

152 template <typename PointT>
153 Eigen::Vector3d GetCloudBarycenter(
154  typename pcl::PointCloud<PointT>::Ptr cloud) {
155  int point_num = cloud->points.size();
156  Eigen::Vector3d barycenter(0, 0, 0);
157 
158  for (int i = 0; i < point_num; i++) {
159  const PointT& pt = cloud->points[i];
160  barycenter[0] += pt.x;
161  barycenter[1] += pt.y;
162  barycenter[2] += pt.z;
163  }
164 
165  if (point_num > 0) {
166  barycenter[0] /= point_num;
167  barycenter[1] /= point_num;
168  barycenter[2] /= point_num;
169  }
170  return barycenter;
171 }
