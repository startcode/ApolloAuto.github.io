132 template <typename SensorType>
133 Status SensorCanbus<SensorType>::Init() {
134  // load conf
135  if (!::apollo::common::util::GetProtoFromFile(FLAGS_sensor_conf_file,
136  &canbus_conf_)) {
137  return OnError("Unable to load canbus conf file: " +
138  FLAGS_sensor_conf_file);
139  }
140 
141  AINFO << "The canbus conf file is loaded: " << FLAGS_sensor_conf_file;
142  ADEBUG << "Canbus_conf:" << canbus_conf_.ShortDebugString();
143 
144  // Init can client
145  auto *can_factory = CanClientFactory::instance();
146  can_factory->RegisterCanClients();
147  can_client_ = can_factory->CreateCANClient(canbus_conf_.can_card_parameter());
148  if (!can_client_) {
149  return OnError("Failed to create can client.");

151  AINFO << "Can client is successfully created.";
152 
153  sensor_message_manager_ = std::unique_ptr<canbus::MessageManager<SensorType>>(

156  return OnError("Failed to create message manager.");
157  }
158  AINFO << "Sensor message manager is successfully created.";
159 
160  if (can_receiver_.Init(can_client_.get(), sensor_message_manager_.get(),
161  canbus_conf_.enable_receiver_log()) != ErrorCode::OK) {
162  return OnError("Failed to init can receiver.");
163  }
164  AINFO << "The can receiver is successfully initialized.";
165 
166  AdapterManager::Init(FLAGS_adapter_config_filename);
167 
168  AINFO << "The adapter manager is successfully initialized.";
169 
170  return Status::OK();
171 }

173 template <typename SensorType>
174 Status SensorCanbus<SensorType>::Start() {
175  // 1. init and start the can card hardware
176  if (can_client_->Start() != ErrorCode::OK) {
177  return OnError("Failed to start can client");
178  }
179  AINFO << "Can client is started.";
180 
181  // 2. start receive first then send
182  if (can_receiver_.Start() != ErrorCode::OK) {
183  return OnError("Failed to start can receiver.");
184  }
185  AINFO << "Can receiver is started.";
186 
187  // 3. set timer to trigger publish info periodically
188  // if sensor_freq == 0, then it is event-triggered publishment.
189  // no need for timer.
190  if (FLAGS_sensor_freq > 0) {
191  const double duration = 1.0 / FLAGS_sensor_freq;
192  timer_ = AdapterManager::CreateTimer(
193  ros::Duration(duration), &SensorCanbus<SensorType>::OnTimer, this);
194  } else {
195  data_trigger_running_ = true;
196  thread_.reset(new std::thread([this] { DataTrigger(); }));
197  if (thread_ == nullptr) {
198  AERROR << "Unable to create data trigger thread.";
199  return OnError("Failed to start data trigger thread.");
200  }
201  }
202 
203  // last step: publish monitor messages
204  apollo::common::monitor::MonitorBuffer buffer(&monitor_);
205  buffer.INFO("Canbus is started.");
206 
207  return Status::OK();
208 }

216 void SensorCanbus<SensorType>::DataTrigger() {
217  std::condition_variable* cvar = sensor_message_manager_->GetMutableCVar();
218  while (data_trigger_running_) {
219  std::unique_lock<std::mutex> lock(mutex_);
220  cvar->wait(lock);

222  AINFO << "===== Publish Sensor Data =====";
223  PublishSensorData();
