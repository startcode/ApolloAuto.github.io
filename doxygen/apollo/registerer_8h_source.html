17 // Registerer is a factory register mechanism that make class registration at
18 // compile time and allow generating an object by giving the name. Example:
19 //
20 // class BaseClass { // base class
21 // ...
22 // };
23 // REGISTER_REGISTERER(BaseClass);
24 // #define REGISTER_BASECLASS(name) REGISTER_CLASS(BaseClass, name)
25 //
26 // class Sub1 : public BaseClass {
27 // ...
28 // };
29 // REGISTER_BASE(Sub1);
30 // class Sub2 : public BaseClass {
31 // ...
32 // };
33 // REGISTER_BASE(Sub2);
34 //
35 // Note that REGISTER_BASE(sub1) should be put in cc file instead of h file,
36 // to avoid multi-declaration error when compile.
37 //
38 // Then you could get a new object of the sub class by:
39 // Base *obj = BaseClassRegisterer::GetInstanceByName("Sub1");
40 //
41 // This is convenient when you need decide the class at runtime or by flag:
42 // string name = "Sub1";
43 // if (...)
44 // name = "Sub2";
45 // Base *obj = BaseClassRegisterer::GetInstanceByName(name);
46 //
47 // If there should be only one instance in the program by desgin,
48 // GetUniqInstance could be used:
49 // Base *obj = BaseClassRegisterer::GetUniqInstance();
50 // If multi sub classes are registered, this method will cause a CHECK fail.
51 //
